140 void customPDE<dim,degree>::residualRHS(variableContainer<dim,degree,dealii::VectorizedArray<double> > & variable_list,
141  dealii::Point<dim, dealii::VectorizedArray<double> > q_point_loc) const {
142 
143 
144 dealii::VectorizedArray<double> fnV = constV(0.0);
145 scalarvalueType ni, nj;
146 scalargradType nix;
147 
148 // In this application, create temporary variables for the residual terms. We cannot
149 // call 'set_scalar_value_residual_term' and 'set_scalar_gradient_residual_term'in the
150 // for loop below because those functions write over the scalar value and scalar gradient
151 // internal variables in 'variable_list' (for performance reasons). Therefore, we wait
152 // to set the residual terms until all the residuals have been calculated.
153 
154 std::vector<scalarvalueType> value_residuals;
155 value_residuals.resize(userInputs.number_of_variables);
156 std::vector<scalargradType> gradient_residuals;
157 gradient_residuals.resize(userInputs.number_of_variables);
158 
159 for (unsigned int i=0; i<userInputs.number_of_variables; i++){
160 
161  ni = variable_list.get_scalar_value(i);
162  nix = variable_list.get_scalar_gradient(i);
163  fnV = - ni + ni*ni*ni;
164  for (unsigned int j=0; j<userInputs.number_of_variables; j++){
165  if (i != j){
166  nj = variable_list.get_scalar_value(j);
167  fnV += constV(2.0*alpha) * ni * nj*nj;
168  }
169  }
170  value_residuals[i] = ni-constV(userInputs.dtValue*MnV)*fnV;
171  gradient_residuals[i] = constV(-userInputs.dtValue*KnV*MnV)*nix;
172 }
173 
174 for (unsigned int i=0; i<userInputs.number_of_variables; i++){
175  variable_list.set_scalar_value_residual_term(i,value_residuals[i]);
176  variable_list.set_scalar_gradient_residual_term(i,gradient_residuals[i]);
177 }
178 
179 }
