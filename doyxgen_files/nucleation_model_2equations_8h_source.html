42 // The model free energy equations and expressions for the residual equations
43 // can be set here. For simple cases, the entire residual equation can be written
44 // here. For more complex cases with loops or conditional statements, residual
45 // equations (or parts of residual equations) can be written below in "residualRHS".
46 
47 // Free energy for each phase and their first and second derivatives
48 #define faV (A0+A2*(c_alpha-calmin)*(c_alpha-calmin))
49 #define facV (2.0*A2*(c_alpha-calmin))
50 #define faccV (2.0*A2)
51 #define fbV (B0+B2*(c_beta-cbtmin)*(c_beta-cbtmin))
52 #define fbcV (2.0*B2*(c_beta-cbtmin))
53 #define fbccV (2.0*B2)
54 
55 // Interpolation function and its derivative
56 #define hV (3.0*n*n - 2.0*n*n*n)
57 #define hnV (6.0*n - 6.0*n*n)
58 
59 // KKS model c_alpha and c_beta as a function of c and h
60 #define c_alpha ((B2*(c-cbtmin*hV) + A2*calmin*hV)/(A2*hV+B2*(1.0-hV)))
61 #define c_beta ((A2*(c-calmin*(1.0-hV))+B2*cbtmin*(1.0-hV))/(A2*hV+B2*(1.0-hV)))
62 
63 // Double-Well function (can be used to tune the interfacial energy)
64 #define fbarrierV (n*n - 2.0*n*n*n + n*n*n*n)
65 #define fbarriernV (2.0*n - 6.0*n*n + 4.0*n*n*n)
66 
67 // Residual equations
68 // For concentration
69 #define term_muxV (cx + (c_alpha - c_beta)*hnV*nx)
70 #define rcV (c)
71 #define rcxV (constV(-McV*userInputs.dtValue)*term_muxV)
72 //For order parameter (gamma is a variable order parameter mobility factor)
73 #define rnV (n-constV(userInputs.dtValue*MnV)*gamma*((fbV-faV)*hnV - (c_beta-c_alpha)*fbcV*hnV + W_barrier*fbarriernV))
74 #define rnxV (constV(-userInputs.dtValue*KnV*MnV)*gamma*nx)
